function downloadTeamPDF(index) { const team = globalCachedTeamsList[index]; const { jsPDF } = window.jspdf; const doc = new jsPDF(); // --- Document Header --- doc.setFillColor(30, 41, 34); // Slate Accent doc.rect(0, 0, 210, 40, 'F'); doc.setFont("Helvetica", "bold"); doc.setFontSize(22); doc.setTextColor(255, 255, 255); doc.text("COMPETITION METRIC SCORE REPORT", 14, 26); // --- Metadata Information Panel --- doc.setFontSize(10); doc.setFont("Helvetica", "normal"); doc.setTextColor(80, 90, 85); doc.text("Profile Summary Identification", 14, 52); doc.setFont("Helvetica", "bold"); doc.setFontSize(14); doc.setTextColor(30, 41, 34); doc.text(`${team.teamName}`, 14, 59); doc.setFont("Helvetica", "normal"); doc.setFontSize(10); doc.setTextColor(80, 90, 85); doc.text("Assigned Scorer Officer", 110, 52); doc.setFont("Helvetica", "bold"); doc.setFontSize(13); doc.setTextColor(30, 41, 34); doc.text(`${team.scorerName || 'N/A'}`, 110, 59); // Divider doc.setDrawColor(210, 215, 212); doc.line(14, 66, 196, 66); // --- Aggregate Yield Metrics --- doc.setFont("Helvetica", "normal"); doc.setFontSize(10); doc.setTextColor(80, 90, 85); doc.text("Aggregated Standing Performance Yield Score", 14, 76); doc.setFont("Helvetica", "bold"); doc.setFontSize(20); doc.setTextColor(45, 122, 67); // Green branding doc.text(`${team.totalScore || 0} Points`, 14, 86); // --- Itemized Performance Resolution Grid --- doc.setFontSize(12); doc.setTextColor(30, 41, 34); doc.text("Comprehensive Item Matrix Verification Status", 14, 102); // Format individual item structures out cleanly to printable table sets const tableHeaders = [["Sector Tier", "Item Identification", "Resolution State Status"]]; const matrixRows = []; const tiers = ["easy", "average", "difficult"]; tiers.forEach(tier => { for(let i = 1; i <= 5; i++) { const rawVal = team.matrix?.[tier]?.[`item${i}`] || "unscored"; let stateDisplay = "Not Scored Yet"; if(rawVal === "correct") stateDisplay = "✔ CORRECT"; if(rawVal === "wrong") stateDisplay = "✘ WRONG"; if(rawVal === "unanswered") stateDisplay = "⦾ NO ANSWER"; matrixRows.push([ tier.toUpperCase(), `Item 0${i}`, stateDisplay ]); } }); doc.autoTable({ head: tableHeaders, body: matrixRows, startY: 108, theme: 'grid', headStyles: { fillColor: [45, 122, 67], fontStyle: 'bold' }, columnStyles: { 0: { fontStyle: 'bold', width: 45 }, 1: { width: 45 }, 2: { fontStyle: 'bold' } }, didParseCell: function(data) { // Apply corporate text-coloring variants depending on data outputs if (data.column.index === 2) { if (data.cell.text[0].includes("CORRECT")) { data.cell.styles.textColor = [45, 122, 67]; } else if (data.cell.text[0].includes("WRONG")) { data.cell.styles.textColor = [202, 60, 60]; } else if (data.cell.text[0].includes("NO ANSWER")) { data.cell.styles.textColor = [43, 92, 184]; } } } }); // --- Footer Authentication Stamps --- const finalY = doc.lastAutoTable.finalY + 15; doc.setFont("Helvetica", "italic"); doc.setFontSize(8); doc.setTextColor(130, 135, 132); doc.text(`Report processed automatically via Live Core Ledger Distribution Framework.`, 14, finalY); doc.save(`Performance_Report_${team.teamName.replace(/\s+/g, '_')}.pdf`); writeLog(`[EXPORT] Processed unique performance sheet for ${team.teamName}`); }
